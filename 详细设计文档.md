#### 2.3.3 ExcelParser（Excel解析器）

```typescript
class ExcelParser implements FileParser {
  extensions = ['.xlsx', '.xls', '.csv'];
  
  async parse(content: Buffer, filename: string): Promise<ParsedItem[]> {
    const workbook = XLSX.read(content);
    const items: ParsedItem[] = [];
    
    for (const sheetName of workbook.SheetNames) {
      const sheet = workbook.Sheets[sheetName];
      const data = XLSX.utils.sheet_to_json(sheet);
      
      // 将每行转为文本
      const rows = data.map((row, idx) => {
        const fields = Object.entries(row)
          .map(([k, v]) => `${k}: ${v}`)
          .join(', ');
        return `第${idx+1}行: ${fields}`;
      });
      
      items.push({
        title: `${filename} - ${sheetName}`,
        content: rows.join('\n'),
        metadata: { sheetName, rowCount: data.length },
        modality: 'table'
      });
    }
    return items;
  }
}
```

### 4.3 向量检索算法

#### 4.3.1 余弦相似度计算

```typescript
function cosineSimilarity(a: number[], b: number[]): number {
  if (a.length !== b.length) {
    throw new Error('Vector dimensions must match');
  }
  
  let dotProduct = 0;
  let normA = 0;
  let normB = 0;
  
  for (let i = 0; i < a.length; i++) {
    dotProduct += a[i] * b[i];
    normA += a[i] * a[i];
    normB += b[i] * b[i];
  }
  
  normA = Math.sqrt(normA);
  normB = Math.sqrt(normB);
  
  if (normA === 0 || normB === 0) {
    return 0;
  }
  
  return dotProduct / (normA * normB);
}
```

#### 4.3.2 向量检索SQL（RPC函数）

```sql
CREATE OR REPLACE FUNCTION vector_search(
  query_embedding VECTOR(2048),
  match_threshold FLOAT DEFAULT 0.3,
  match_count INT DEFAULT 20
)
RETURNS TABLE (
  id UUID,
  title TEXT,
  content TEXT,
  source TEXT,
  modality TEXT,
  similarity FLOAT
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT
    ki.id,
    ki.title,
    ki.content,
    ki.source,
    ki.modality,
    1 - (ki.embedding <=> query_embedding) AS similarity
  FROM knowledge_items ki
  WHERE ki.embedding IS NOT NULL
    AND 1 - (ki.embedding <=> query_embedding) > match_threshold
  ORDER BY ki.embedding <=> query_embedding
  LIMIT match_count;
END;
$$;
```

```typescript
async function comprehensiveSearch(query: string): Promise<SearchResult[]> {
  const results: SearchResult[] = [];
  
  // 1. 查询向量化
  const queryEmbedding = await embeddingService.embedOne(query);
  
  // 2. 搜索knowledge_items表
  const knowledgeResults = await searchKnowledgeItems(queryEmbedding);
  results.push(...knowledgeResults);
  
  // 3. 搜索port_data表
  const portResults = await searchPortData(queryEmbedding);
  results.push(...portResults.map(r => ({
    ...r,
    modality: 'port',
    source: 'port_data'
  })));
  
  // 4. 搜索route_data表
  const routeResults = await searchRouteData(queryEmbedding);
  results.push(...routeResults.map(r => ({
    ...r,
    modality: 'route',
    source: 'route_data'
  })));
  
  // 5. 按相似度排序
  results.sort((a, b) => b.similarity - a.similarity);
  
  // 6. 返回TopN
  return results.slice(0, 20);
}
```

### 5.5 流式输出实现

#### 后端实现

```typescript
// API Route
export async function POST(request: Request) {
  const { question } = await request.json();
  
  // 创建ReadableStream
  const stream = new ReadableStream({
    async start(controller) {
      const encoder = new TextEncoder();
      
      try {
        // 1. 检索
        const results = await searchEngine.search(question);
        controller.enqueue(encoder.encode(
          `data: ${JSON.stringify({ type: 'search', results })}\n\n`
        ));
        
        // 2. LLM流式推理
        const llmStream = await llmService.chatStream({
          messages: buildPrompt(formatResults(results), question)
        });
        
        const reader = llmStream.getReader();
        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
          
          controller.enqueue(encoder.encode(
            `data: ${JSON.stringify({ type: 'content', text: value })}\n\n`
          ));
        }
        
        controller.enqueue(encoder.encode(
          `data: ${JSON.stringify({ type: 'done' })}\n\n`
        ));
      } catch (error) {
        controller.enqueue(encoder.encode(
          `data: ${JSON.stringify({ type: 'error', message: error.message })}\n\n`
        ));
      }
      
      controller.close();
    }
  });
  
  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive'
    }
  });
}
```

```typescript
async function askQuestion(question: string) {
  const response = await fetch('/api/rag', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ question })
  });
  
  const reader = response.body!.getReader();
  const decoder = new TextDecoder();
  let answer = '';
  
  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    
    const chunk = decoder.decode(value);
    const lines = chunk.split('\n').filter(line => line.startsWith('data: '));
    
    for (const line of lines) {
      const data = JSON.parse(line.slice(6));
      
      if (data.type === 'content') {
        answer += data.text;
        setAnswer(answer);  // 打字机效果
      } else if (data.type === 'search') {
        setSearchResults(data.results);
      }
    }
  }
}
```

```typescript
function parseCSVLine(line: string): string[] {
  const result: string[] = [];
  let current = '';
  let inQuotes = false;
  
  for (let i = 0; i < line.length; i++) {
    const char = line[i];
    
    if (char === '"') {
      if (inQuotes && line[i + 1] === '"') {
        // 双引号转义
        current += '"';
        i++;
      } else {
        // 切换引号状态
        inQuotes = !inQuotes;
      }
    } else if (char === ',' && !inQuotes) {
      // 字段分隔
      result.push(current.trim());
      current = '';
    } else {
      current += char;
    }
  }
  
  result.push(current.trim());
  return result;
}
```

```typescript
  { name: '北美', bounds: [[25, -130], [70, -60]] },
  { name: '中南美', bounds: [[-60, -90], [25, -30]] }
];

function countByRegion(ports: PortData[]): Record<string, number> {
  const counts: Record<string, number> = { '全球': ports.length };
  
  for (const region of REGIONS) {
    const [[lat1, lon1], [lat2, lon2]] = region.bounds;
    counts[region.name] = ports.filter(p => 
      p.lat >= lat1 && p.lat <= lat2 &&
      p.lon >= lon1 && p.lon <= lon2
    ).length;
  }
  
  return counts;
}
```

### 8.3 文档切分算法

```typescript
interface ChunkConfig {
  chunkSize: number;      // 切分大小（字符）
  chunkOverlap: number;   // 重叠大小
  separator: string;      // 分隔符
}

const DEFAULT_CHUNK_CONFIG: ChunkConfig = {
  chunkSize: 500,
  chunkOverlap: 50,
  separator: '\n\n'
};

function splitDocument(content: string, config: ChunkConfig): string[] {
  const { chunkSize, chunkOverlap, separator } = config;
  const chunks: string[] = [];
  
  // 先按分隔符分割
  const paragraphs = content.split(separator);
  
  let currentChunk = '';
  
  for (const para of paragraphs) {
    if ((currentChunk + para).length > chunkSize) {
      if (currentChunk) {
        chunks.push(currentChunk);
        // 保留重叠部分
        currentChunk = currentChunk.slice(-chunkOverlap);
      }
    }
    currentChunk += (currentChunk ? separator : '') + para;
  }
  
  if (currentChunk) {
    chunks.push(currentChunk);
  }
  
  return chunks;
}
```

```typescript
function useVirtualScroll<T>(
  items: T[],
  config: VirtualScrollConfig
) {
  const { itemHeight, containerHeight, overscan } = config;
  const [scrollTop, setScrollTop] = useState(0);
  
  const startIndex = Math.max(0, Math.floor(scrollTop / itemHeight) - overscan);
  const endIndex = Math.min(

  );
  
  const visibleItems = items.slice(startIndex, endIndex);
  const offsetY = startIndex * itemHeight;
  
  return { visibleItems, offsetY, totalHeight: items.length * itemHeight };
}
```

```typescript
function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeoutId: NodeJS.Timeout | null = null;
  
  return (...args: Parameters<T>) => {
    if (timeoutId) {
      clearTimeout(timeoutId);
    }
    timeoutId = setTimeout(() => func(...args), wait);
  };
}

function throttle<T extends (...args: any[]) => any>(
  func: T,
  limit: number
): (...args: Parameters<T>) => void {
  let inThrottle = false;
  
  return (...args: Parameters<T>) => {
    if (!inThrottle) {
      func(...args);
      inThrottle = true;
      setTimeout(() => { inThrottle = false; }, limit);
    }
  };
}
```

```typescript
class CacheService {
  private cache = new Map<string, { value: any; expireAt: number }>();
  
  get<T>(key: string): T | null {
    const item = this.cache.get(key);
    if (!item || Date.now() > item.expireAt) {
      this.cache.delete(key);
      return null;
    }
    return item.value as T;
  }
  
  set<T>(key: string, value: T, ttl: number): void {
    this.cache.set(key, {
      value,
      expireAt: Date.now() + ttl
    });
  }
}

// 使用示例
```
